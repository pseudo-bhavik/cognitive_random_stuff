    <script>
        // --- POINTS TOGGLE LOGIC ---
        let isMeshVisible = false;

        function toggleMesh() {
            isMeshVisible = !isMeshVisible;
            const btn = document.getElementById('mesh-toggle-btn');
            
            if(isMeshVisible) {
                btn.innerText = "DIAGNOSTIC OVERLAY: ON";
                btn.classList.add('bg-black', 'text-white');
            } else {
                btn.innerText = "DIAGNOSTIC OVERLAY: OFF";
                btn.classList.remove('bg-black', 'text-white');
            }

            fetch('/api/toggle_mesh', {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({ show_mesh: isMeshVisible })
            }).catch(err => console.error("Toggle Error:", err));
        }

        // --- CHART & METRICS LOGIC ---
        const ctx = document.getElementById('telemetryChart').getContext('2d');
        const MAX_DATA_POINTS = 50; 

        const telemetryChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: Array(MAX_DATA_POINTS).fill(''), 
                datasets: [
                    {
                        label: 'MOE (Fatigue Ratio)',
                        data: Array(MAX_DATA_POINTS).fill(null),
                        borderColor: '#000000',
                        borderWidth: 3,
                        pointRadius: 0,
                        tension: 0.1,
                        yAxisID: 'y'
                    },
                    {
                        label: 'EAR (Focus Drop)',
                        data: Array(MAX_DATA_POINTS).fill(null),
                        borderColor: '#aaaaaa',
                        borderDash: [5, 5],
                        borderWidth: 2,
                        pointRadius: 0,
                        tension: 0.1,
                        yAxisID: 'y1'
                    }
                ]
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                animation: false, 
                plugins: {
                    legend: { labels: { color: '#000', font: { family: 'ui-monospace, monospace', weight: 'bold' } } }
                },
                scales: {
                    x: { display: false }, 
                    y: { 
                        type: 'linear', display: true, position: 'left',
                        grid: { color: '#e5e5e5' },
                        ticks: { color: '#000', font: { family: 'ui-monospace, monospace' } },
                        title: { display: true, text: 'MOE', font: { weight: 'bold' } }
                    },
                    y1: {
                        type: 'linear', display: true, position: 'right',
                        grid: { drawOnChartArea: false },
                        ticks: { color: '#aaaaaa', font: { family: 'ui-monospace, monospace' } },
                        title: { display: true, text: 'EAR', color: '#aaaaaa', font: { weight: 'bold' } }
                    }
                }
            }
        });

        function updateMetrics() {
            fetch('/api/metrics')
                .then(response => {
                    if (response.status === 401) {
                        window.location.href = '/login';
                        throw new Error('Unauthorized');
                    }
                    return response.json();
                })
                .then(data => {
                    // Update Vision Text
                    document.getElementById('val-ear').innerText = data.ear.toFixed(2);
                    document.getElementById('val-mar').innerText = data.mar.toFixed(2);
                    document.getElementById('val-moe').innerText = data.moe.toFixed(2);
                    document.getElementById('val-blinks').innerText = data.blinks;
                    
                    // Update Status Badge
                    const statusEl = document.getElementById('status-indicator');
                    statusEl.innerText = data.status;
                    if(data.status.includes("FATIGUE")) {
                        statusEl.style.backgroundColor = "#fff";
                        statusEl.style.color = "#000";
                        statusEl.style.border = "4px solid #000";
                    } else {
                        statusEl.style.backgroundColor = "#000";
                        statusEl.style.color = "#fff";
                        statusEl.style.border = "none";
                    }

                    // --- NEW: DYNAMIC HARDWARE UPDATE ---
                    const gsrEl = document.getElementById('val-gsr');
                    if (data.gsr !== "OFFLINE") {
                        gsrEl.innerText = data.gsr + " RAW";
                        gsrEl.classList.remove('bg-black', 'text-white');
                        gsrEl.classList.add('border-heavy', 'bg-white', 'text-black', 'px-3');
                    }

                    // Update Chart Data
                    const chartData = telemetryChart.data.datasets;
                    chartData[0].data.shift();
                    chartData[0].data.push(data.moe);
                    
                    chartData[1].data.shift();
                    chartData[1].data.push(data.ear);
                    
                    telemetryChart.update();
                })
                .catch(err => console.error("Data polling error:", err));
        }

        setInterval(updateMetrics, 1000);
    </script>
